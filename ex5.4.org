Recursive exponentiation:

#+begin_src scheme
  (controller
   (assign continue expt-done)
   expt-loop
   (test (op =) (reg n) (const 0))
   (branch immediate-answer)
   (save continue)
   (assign n (op -) (reg n) (const 1))
   (goto (label expt-loop))
   (restore continue)
   (assign n (op *) (reg b) (reg n))
   (goto continue)
   immedidate-answer
   (assign n (const 1))
   (goto (reg continue))
   expt-done)
#+end_src
